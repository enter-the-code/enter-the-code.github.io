# Client.c

아래는 클라이언트에 관한한 코드이며 처음 소캣으로 서버와 에코 통신을 해보는 코드이다.
</br>

통신의 순서는 아래와 같다

![소켓 통신 순서](/media/tcp_order.png)

```Cpp
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <arpa/inet.h>

int main(int argc,char** argv){
    int sockfd; // 사용할 소켓 
    char buf[1024];
    char hello[100];
    struct sockaddr_in servaddr;// 주소를 담을 구조체
    //만약에 포트번호 그리고 아이피주소를 전부 매개변수로 주지않을 시 오류를 출력하는 부분
    if (argc<3)
    {
        /* code */
        printf("usage:./client remoteAddress remotePort\n");
        return -1;
    }
    //TCP로 소켓생성 "SOCK_STREAM"
    if ((sockfd=socket(AF_INET,SOCK_STREAM,0))<0)
    {
        /* code */
        perror("socket creation failed\n");
        return -1;
    }
    //주소 입력
    servaddr.sin_family=AF_INET;
    servaddr.sin_addr.s_addr=inet_addr(argv[2]);
    servaddr.sin_port=htons(atoi(argv[1]));
    //connect
    if(connect(sockfd,(const struct sockaddr*)&servaddr,sizeof(servaddr))<0)
    {
        perror("connect error\n");
        return -1;
    }
    //에코 형태로 줬던메시지 다시 돌려받음
    memset(buf,0,sizeof(buf));
    memset(hello,0,sizeof(hello));
    scanf("%s",hello);
    write(sockfd,hello,strlen(hello));
    read(sockfd,buf,sizeof(buf));
    printf("messege from server:%s\n",buf);
    close(sockfd);
    return 0;

}
```

# Server.c
아래는 서버 코드이다.

```cpp
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/socket.h>
#include <arpa/inet.h>

typedef struct sockaddr_in sok_in;

int main(int argc, char** argv)
{
    int sockfd,cSockfd;
    sok_in servaddr,cliaddr;
    char buf[1024];
    socklen_t len;
    //포트번호 안받으면 오류
    if (argc<2)
    {
        /* code */
        printf("usage:./server localPort\n");
        return -1;
    }
    //소켓 열기
    if ((sockfd=socket(AF_INET,SOCK_STREAM,0))<0)
    {
        /* code */
        perror("socket creation failed\n");
        return -1;
    }
    //소켓 옵션 설정
    int enable=1;
    setsockopt(sockfd,SOL_SOCKET,SO_REUSEADDR,&enable,sizeof(int));

    servaddr.sin_family=AF_INET;
    servaddr.sin_addr.s_addr=htonl(INADDR_ANY);
    servaddr.sin_port=htons(atoi(argv[1]));
    //주소랑 바인딩(묶음)
    if (bind(sockfd,(struct sockaddr*)&servaddr,sizeof(servaddr))<0)
    {
        /* code */
        perror("bind failed\n");
        return -1;
    }
    //listen 5명의 대기자
    if (listen(sockfd,5)<0)
    {
        /* code */
        perror("socket failed\n");
        return -1;
    }
    //accept 받아드릴 준비가 된상태
    if ((cSockfd=accept(sockfd,(struct sockaddr*)&cliaddr,&len))<0)
    {
        /* code */
        perror("accept error");
        return -1;
    }
    //받았던 부분에 이름만 붙여서 다시 클라이언트로 보냄
    memset(buf,0,sizeof(buf));
    read(cSockfd,buf,sizeof(buf));
    printf("%s\n",buf);

    strcat(buf,"_이름");
    write(cSockfd,buf,strlen(buf));


    
    close(cSockfd);
    close(sockfd);

}
```
